$repoPath = "D:\note\gmgn3022"
$debounceSeconds = 6

$watcher = New-Object System.IO.FileSystemWatcher
$watcher.Path = $repoPath
$watcher.Filter = "*.*"
$watcher.IncludeSubdirectories = $true
$watcher.EnableRaisingEvents = $true

$global:lastEventTime = Get-Date
$global:syncRunning = $false

function Should-IgnoreFile($filePath) {
    if ($filePath -match '\\\.git\\') { return $true }
    if ($filePath -match '\\node_modules\\') { return $true }
    if ($filePath -match '\\\.obsidian\\') { return $true }
    if ($filePath -match '\.tmp$') { return $true }
    if ($filePath -match '\.swp$') { return $true }
    if ($filePath -match '~$') { return $true }
    return $false
}

function Run-GitSync {
    if ($global:syncRunning) { return }
    $global:syncRunning = $true
    try {
        Set-Location $repoPath
        git add .
        git diff --cached --quiet
        if ($LASTEXITCODE -eq 0) { Write-Host "$(Get-Date -Format 'HH:mm:ss') No changes to commit"; return }
        $msg = "auto update $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss')"
        git commit -m $msg
        if ($LASTEXITCODE -ne 0) { Write-Host "$(Get-Date -Format 'HH:mm:ss') Commit failed"; return }
        git pull --rebase
        if ($LASTEXITCODE -ne 0) { Write-Host "$(Get-Date -Format 'HH:mm:ss') Pull --rebase failed, need manual resolve"; return }
        git push
        if ($LASTEXITCODE -eq 0) { Write-Host "$(Get-Date -Format 'HH:mm:ss') Push success" }
        else { Write-Host "$(Get-Date -Format 'HH:mm:ss') Push failed" }
    }
    finally { $global:syncRunning = $false }
}

$action = {
    $filePath = $Event.SourceEventArgs.FullPath
    if (Should-IgnoreFile $filePath) { return }
    $global:lastEventTime = Get-Date
    Write-Host "$(Get-Date -Format 'HH:mm:ss') Change detected: $filePath"
}

Register-ObjectEvent $watcher Changed -Action $action | Out-Null
Register-ObjectEvent $watcher Created -Action $action | Out-Null
Register-ObjectEvent $watcher Deleted -Action $action | Out-Null
Register-ObjectEvent $watcher Renamed -Action $action | Out-Null

Write-Host "Watching $repoPath"
Write-Host "After saving a file, idle $debounceSeconds seconds will auto-sync"
Write-Host "Press Ctrl+C to exit"

while ($true) {
    Start-Sleep -Seconds 2
    $idleSeconds = ((Get-Date) - $global:lastEventTime).TotalSeconds
    if ($idleSeconds -ge $debounceSeconds) { Run-GitSync; $global:lastEventTime = Get-Date }
}
